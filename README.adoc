setx HOME "d:\home\path\to\use"

sed -ie '/db_shell:/cdb_shell: /usr/bin/zsh' /etc/nsswitch.conf
sed -ie "/db_home:/cdb_home: $HOME" /etc/nsswitch.conf

# restart
